class Solution(object):
    def myPow(self, x, n):
        if n == 0:
            return 1
        
        # handle negative power
        if n < 0:
            x = 1 / x
            n = -n
        
        result = 1
        
        while n > 0:
            if n % 2 == 1:   # odd
                result *= x
            
            x *= x
            n //= 2
        
        return result
